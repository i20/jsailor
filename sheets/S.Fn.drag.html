JSailor S.Fn.drag(Object options, Function callback(Point delta))

Description:

  • Rend les éléments ciblés déplaçables à la souris.

Remarques:

  • Les 2 paramètres sont facultatifs.

Exemples: 

  • <!DOCTYPE html>
<html>
    <head>
        <script type="text/javascript" src="jsailor-1.1.5.min.js"></script>
        <script type="text/javascript">
        S(function(){

            var divs = S('div');

            divs.item(0).drag();
            divs.item(1).drag(function(delta){

                alert('J\'ai bougé d\'un vecteur (' + delta.x + ', ' + delta.y + ')');
            });
            divs.item(2).drag({ method: 'ghost' });

        });
        </script>
        <style type="text/css">
        div
        {
            background-color: red;
            height: 30px;
            width: 30px;
            position: absolute;
        }
        </style>
    </head>
    <body>
        <div style="top:0px">1</div>
        <div style="top:40px">2</div>
        <div style="top:80px">3</div>
    </body>
</html>
