Exemples simples d'utilisation

Nous allons voir ci-dessous quelques exemples concrets de ce que vous permet de faire JSailor.

<head>
    <script type="text/javascript" src="ici l'adresse du fichier de la librairie"></script>
    <script type="text/javascript">

        var reset = function(){

            var but = S('#button4');

            but.youngers().remove();

            but.follow('
Je suis le paragraphe qui va subir les changements!
');
        };

        S(function(){

            reset();

            S('#button1').click(function(){

                S(this).siblings('p').html( prompt('Tapez le nouveau contenu du paragraphe') );
            });

            S('#button2').click(function(){

                S(this).siblings('p').tune({

                    'color': 'orange',
                    'font-size': '3em'

                }, 1000);
            });

            S('#button3').click(function(){

                S(this).next().follow( S(this).siblings('p').clone() );
            });

            S('#button4').click(reset);
        });

    </script>
</head>
