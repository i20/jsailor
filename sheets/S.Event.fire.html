Void S.Event.fire(Element elt, String type, Event event)

Description:

  • Déclenche l'événement 
    type
    ayant les caractèristiques 
    event
    (cf objet event) sur 
    elt
    .

Remarques: 

  • type
    correspond au type d'événement non préfixé par 'on'.

Exemples: 

  • <html>
    <head>
        <script type="text/javascript" src="jsailor-1.1.4.min.js"></script>
        <script type="text/javascript">
        S(function(){

            var paras = document.getElementsByTagName('p');

            //On attache tout d'abord un événement au premier paragraphe:
            S.Event.bind(paras[0], 'click', function(){

                alert( this.innerHTML );
            });

            //On déclenche ensuite l'événement attaché précédemment:
            S.Event.fire(paras[0], 'click');
        });
        </script>
    </head>
    <body>
        <div>
            <p>Cliquez sur moi!</p>
            <p>Je suis le second paragraphe.</p>
        </div>
        <div>
            <p>Je suis le troisième paragraphe.</p>
        </div>
    </body>
</html>
  • <html>
    <head>
        <script type="text/javascript" src="jsailor-1.1.4.min.js"></script>
        <script type="text/javascript">
        S(function(){

            var paras = document.getElementsByTagName('p');

            //On attache tout d'abord un événement au premier paragraphe:
            S.Event.bind(paras[0], 'mouseover', function(event){

                alert( event.pageX );
            });

            //On déclenche ensuite l'événement attaché précédemment en lui ajoutant des caractèristiques:
            S.Event.fire(paras[0], 'mouseover', {

                pageX: 120,
                pageY: 30
            });
        });
        </script>
    </head>
    <body>
        <div>
            <p>Passez la souris sur moi!</p>
            <p>Je suis le second paragraphe.</p>
        </div>
        <div>
            <p>Je suis le troisième paragraphe.</p>
        </div>
    </body>
</html>
