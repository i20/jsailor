Object S.Object.extend(Object base, Object rajout)

Description:

  • Renvoie 
    base
    étendu avec les propriétés de 
    rajout
    .

Remarques:

  • Les propriétés de 
    rajout
    déjà présentes dans 
    base
    seront réécrites.

Exemples: 

  • var test1 = S.Object.extend({

    prop1: 'un',
    prop2: '2',
    prop4: 4

}, {

    prop3: 3,
    prop5: function(arg){

         alert(arg);
    }
});

/*test1 = {

    prop1: 'un',
    prop2: '2',
    prop3: 3,
    prop4: 4,
    prop5: function(arg){

        alert(arg);
    }
}*/
  • var test2 = S.Object.extend({

    prop1: 'un',
    prop2: '2'

}, {

    prop3: 3,
    prop2: function(arg){

        alert(arg);
    }
});

//prop2 a été réécrite
/*test2 = {

    prop1: 'un',
    prop2: function(arg){

        alert(arg);
    },
    prop3: 3

}*/
