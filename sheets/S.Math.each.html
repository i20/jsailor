Void S.Math.each(Number nb, Function fn)

Description:

  • Applique 
    fn
    à chaque chiffre composant 
    nb
    .

Remarques:

  • Dans la fonction, 
    this
    est le chiffre courant.
  • La fonction peut prendre 2 arguments: la puissance de dix pondérant le chiffre courant dans 
    nb
    ; le chiffre courant(idem 
    this
    ).
  • Si 
    nb
    est négatif, les chiffres le composant seront tous négatifs.
  • Les chiffres sont traités dans l'ordre des puissances de dix décroissantes.

Exemples: 

  • var calcul = '';

S.Math.each(4408, function(i, n){

    calcul += (calcul ? ' + ' : '') + n + '*10^' + i;
});

document.write(calcul);
//affiche '4*10^3 + 4*10^2 + 0*10^1 + 8*10^0'

  • var calcul = '';

S.Math.each(0.57, function(i, n){

    calcul += (calcul ? ' + ' : '') + n + '*10^' + i;
});

document.write(calcul);
//affiche '0*10^0 + 5*10^-1 + 7*10^-2'

  • var calcul = '';

S.Math.each(-4408, function(i, n){

    calcul += (calcul ? ' + ' : '') + n + '*10^' + i;
});

document.write(calcul);
//affiche '-4*10^3 + -4*10^2 + 0*10^1 + -8*10^0'
